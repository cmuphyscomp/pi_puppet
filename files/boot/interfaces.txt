# /boot/interfaces.txt
# symbolically linked from /etc/networks/interfaces -> /boot/interfaces.txt

# this controls the network configuration under Linux on the Raspberry Pi 2

auto lo

iface lo inet loopback

# original wired ethernet configuration based on DHCP
# iface eth0 inet dhcp

# wired ethernet configuration for a static IP address.
iface eth0 inet static
   address 192.168.2.3
   netmask 255.255.255.0
   gateway 192.168.2.1
   dns-nameservers 128.2.1.10
   dns-search local

allow-hotplug wlan0
iface wlan0 inet manual
wpa-roam /etc/wpa_supplicant/wpa_supplicant.conf
iface default inet dhcp
